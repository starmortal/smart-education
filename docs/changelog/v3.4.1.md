```javascript
// 根据选中日期筛选错题
const displayErrors = computed(() => {
  if (!selectedDate.value) {
    return errorList.value; // 显示全部
  }
  
  // 筛选选中日期的错题
  return errorList.value.filter(error => {
    const errorDate = dayjs(error.addTime).format('YYYY-MM-DD');
    return errorDate === selectedDate.value;
  });
});
```
